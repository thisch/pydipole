  \mathbf{H}(\mathbf{r}, t) & = & \sum_{n=1}^{N}\frac{ck^{2}}{4\pi r'} (\mathbf{\hat{r}'} \times \mathbf{p}_{n}) \left (1 - \frac{1}{ikr'}\right) e^{ikr' - i(\omega t + \varphi_{n})} \\
